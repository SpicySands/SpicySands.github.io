//@title test @by Dee

const dbank = "rolandtr808"
setCpm(180)

// Drums
$:s("bd(1,4,0.1), sd(<1 2 1 3>,4,1.9)").bank(dbank)._punchcard()
$:s("hh(<1 2>,4,0.1)").gain(.2).sometimesBy(.1, x => x.ply("2"))._punchcard()
// D#F#A# A#C#E# BD#F# G#BD#
// Lead
$:note("< [D#,F#,A#](10,16,0) [A#,C#,E#](16,16,0) [B,D#,F#](10,16,0) [G#,B,D#](16,16,0)>/4")
.gain(.4).trans(36).sound("supersaw")

// Bass
$:note("<[D#,F#,A#] [A#,C#,E#] [B,D#,F#] [G#,B,D#]>/4").trans(-36).gain(.2).sound("square")
