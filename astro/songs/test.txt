sound("<hh(<1 2>,4,0.1)>*4").sometimesBy(.1, x => x.ply("2"))._punchcard(),

// Lead
note("< [D#,F#,A#](10,16,0) [A#,C#,E#](16,16,0) [B,D#,F#](10,16,0) [G#,B,D#](16,16,0)>4")
.trans(36).sound("supersaw"),

// Bass
note("<[D#,F#,A#] [A#,C#,E#] [B,D#,F#] [G#,B,D#]>").trans(-36).sound("square")
)
orbit.postgain(mvs)
